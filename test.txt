\textbf{Eliminate $p$}:\\
We have that 
\begin{align*}
S_{1} = &\{\neg p, q\}, \{\neg q, \neg r, s\}, \{p\}, \{r\}, \{\neg s\}
\end{align*}
Removing any tautologies, we get that 
\begin{align*}
S'_{1} = &\{\neg p, q\}, \{\neg q, \neg r, s\}, \{p\}, \{r\}, \{\neg s\}
\end{align*}
We select clauses with the variable $p$ to get 
\begin{align*}
T_{1} = &\{\neg p, q\}, \{p\}
\end{align*}
We begin resolving clauses in $T_{1}$ to get
\begin{itemize}
\item $\{q\}$ by resolving $\{\neg p, q\}$ and $\{p\}$
\end{itemize}
Therefore, we have that 
\begin{align*}
U_{1} = &\{q\}
\end{align*}
\textbf{Eliminate $q$}:\\
Now, $S_{2} = (S'_{1} \backslash T_{1}) \cup U_{1}$. 
We have that 
\begin{align*}
S_{2} = &\{\neg q, \neg r, s\}, \{r\}, \{\neg s\}, \{q\}
\end{align*}
Removing any tautologies, we get that 
\begin{align*}
S'_{2} = &\{\neg q, \neg r, s\}, \{r\}, \{\neg s\}, \{q\}
\end{align*}
We select clauses with the variable $q$ to get 
\begin{align*}
T_{2} = &\{\neg q, \neg r, s\}, \{q\}
\end{align*}
We begin resolving clauses in $T_{2}$ to get
\begin{itemize}
\item $\{\neg r, s\}$ by resolving $\{\neg q, \neg r, s\}$ and $\{q\}$
\end{itemize}
Therefore, we have that 
\begin{align*}
U_{2} = &\{\neg r, s\}
\end{align*}
\textbf{Eliminate $r$}:\\
Now, $S_{3} = (S'_{2} \backslash T_{2}) \cup U_{2}$. 
We have that 
\begin{align*}
S_{3} = &\{r\}, \{\neg s\}, \{\neg r, s\}
\end{align*}

S'_{3} = &\{r\}, \{\neg s\}, \{\neg r, s\}

Therefore, we have that 
\begin{align*}

\end{align*}
